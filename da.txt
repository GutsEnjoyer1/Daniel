git config --global user.name "Your Name"
git config --global user.email "you@email.com"
git config --global core.editor "code --wait"   # use VS Code instead of Vim

STARTING
git init                          # start a new repo from scratch
git clone <url>                   # download repo from GitHub

DAILY (most used)
git status                        # see what changed
git add .                         # stage all changes
git add filename.py               # stage one file
git commit -m "your message"      # save snapshot
git push                          # upload to GitHub
git pull                          # download latest from GitHub

CHECKING
git log --oneline                 # compact commit history
git log                           # full commit history
git diff                          # see line by line changes
git diff filename.py              # changes in one specific file
git branch                        # list all branches

BRANCHES
git branch name                   # create branch
git checkout name                 # switch to branch
git checkout -b name              # create AND switch in one command
git merge name                    # merge branch into current one
git branch -d name                # delete branch (after merging)
git branch -D name                # force delete branch
git push origin name              # push branch to GitHub

UNDOING MISTAKES
git restore filename.py           # undo changes in a file (before commit)
git restore --staged filename.py  # unstage a file (remove from add)
git reset --soft HEAD~1           # undo last commit (keeps file changes)
git reset --hard HEAD~1           # undo last commit AND delete changes
git merge --abort                 # cancel a merge that went wrong
git revert <commit>               # undo a specific commit safely

STASH (save for later)
git stash                         # hide uncommitted changes temporarily
git stash pop                     # bring them back
git stash list                    # see all stashed changes
git stash drop                    # delete the stash

REMOTE (GitHub connection)
git remote -v                     # see connected GitHub URL
git remote add origin <url>       # connect local repo to GitHub
git push -u origin main           # first push ever (sets up tracking)
git fetch                         # check for changes without downloading

MISCELLANEOUS
git --version                     # check Git is installed
where git                         # see where Git is installed
cd foldername                     # enter a folder
cd ..                             # go back one folder
dir                               # list files in current folder (Windows)
code .                            # open current folder in VS Code
